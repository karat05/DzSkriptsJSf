const silent = document.getElementById("silent");
const xxx = document.getElementById("heal");


const toggleVisibility = () => {
 
    if (xxx.style.display === "none") {
        xxx.style.display = "block"; 
    } else {
        xxx.style.display = "none";
    }
};

silent.addEventListener("click", toggleVisibility);
